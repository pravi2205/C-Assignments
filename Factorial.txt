/*factorial of number using recusrion*/

#include<stdio.h>
int fact(int n) {
    if (n>=1)
        return n*fact(n-1);
    else
        return 1;
}

int main() {
    int n;
    printf("Enter a number: ");
    scanf("%d",&n);
    printf("Factorial of %d = %d", n, fact(n));
    return 0;
}

/*output*/
Enter a number:10
Factorial of 10 =3628800
