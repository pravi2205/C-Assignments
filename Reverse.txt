#include <stdio.h>

int main() {
  int arr[10];
  printf("enter the values in an array:");
  for(int i=0;i<10;i++)
  {
    scanf("%d",&arr[i]);
  }
  int len=sizeof(arr)/sizeof(arr[0]);
  reverse_arr(arr,len);
  printf("array after reverse:");
   for(int i=0;i<10;i++)
  {
    printf("%d\t",arr[i]);
  }
  return 0;
}
void reverse_arr(int arr[10],int len)
{
    int tmp;
    for(int i=0;i<len/2;i++)
    {
        tmp=arr[i];
        arr[i]=arr[len-1-i];
        arr[len-1-i]=tmp;
    }
}
