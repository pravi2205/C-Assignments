
/*Program to explain all operators using different data types*/

#include<stdio.h>  //Header file
int main()
{
int a=10, b=7;   //Initialising int variables
float c=5.4, d=6.8;  //Initialising float variables

//Printing various values
printf("The Addition of %d and %d is: %d\n",a,b,a+b);
printf("The Subtraction of %d and %d is: %d\n",a,b,a-b);
printf("The Multiplication of %d and %d is: %d\n",a,b,a*b);
printf("The Division of %d and %d is: %d\n",a,b,a/b);
printf("The Modulus operation of %d and %d is: %d\n",a,b,a%b);
printf("The Incremented value ++a is: %d\n",++a);
printf("The Decremented value --b is: %d\n",--b);
printf("The Addition of %f and %f is: %f\n",c,d,c+d);
printf("The Subtraction of %f and %f is: %f\n",c,d,c-d);
printf("The Multiplication of %f and %f is: %f\n",c,d,c*d);
printf("The Division of %f and %f is: %f\n",c,d,c/d);
printf("The Modulus operation of %f and %f is: %f\n",c,d,c%d);
printf("The Incremented value ++c is: %f\n",++c);
printf("The Decremented value --d is: %f\n",--d);

return 0;
}
