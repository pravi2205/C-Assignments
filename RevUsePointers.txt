/*Program to find reverse of array using pointers*/

#include<stdio.h>  
  
#define N 5  
  
int main()  
{  
    int a[N], i, *ptr;  
  
    printf("Enter %d  numbers\n", N);  
    for(i = 0; i < N; i++)  
        scanf("%d", &a[i]);  
  
    ptr = &a[N - 1];  
  
    printf("\nElements of array in reverse order \n");  
    for(i = 0; i < N; i++)  
        printf("%d\t", *ptr--);  
  
    return 0;  
}  


/*outout*/

Enter 5 integer numbers
1 2 3 4 5 

Elements of array in reverse order
5       4       3       2       1
